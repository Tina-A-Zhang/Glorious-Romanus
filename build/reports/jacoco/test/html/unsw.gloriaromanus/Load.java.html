package unsw.gloriaromanus;

import java.io.FileInputStream;
import java.io.ObjectInputStream;

public class Load {

    static Game game;
    private Player player;

    /**
     * loading game class
     * @param game the game to load
     * @param player the player who wants to load
     */
    public Load(Game game, Player player) {
        this.player = player;
    }

    public String getPlayerByName(Player player){
        return player.getName();
    }

    /**
     * load the state of the game
     * @throws Exception
     */
    public void LoadGameState() throws Exception {
            String fileName = getPlayerByName(player);
            FileInputStream fis = new FileInputStream(fileName + ".sav");
            ObjectInputStream ois = new ObjectInputStream(fis);
            game = (Game) ois.readObject();
            ois.close();
    }

}
